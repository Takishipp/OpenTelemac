getErrorMASCARET

Gives the error message of a MASCARET instance

Calling sequence

[error,message]=getErrorMASCARET(id)

Arguments

id

MASCARET instance number returned by the function createMASCARET

error

(Output argument) Error code value : zero indicates that no errors occurred

message

Character string indicating the error

Description

Gives the error message of a specific MASCARET instance

Example

toolbox_dir=getenv("toolbox_dir");
c = filesep();

// creation of the MASCARET model
[err, id] = createMASCARET();
assert_checktrue(id>0);

// read data from files
path_xml = "file://"+toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1.xcas";
TabNomFichier = [strsubst(path_xml,'\','/'), ..
                 toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1.geo", ..
                 toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1_0.loi", ..
                 toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1_1.loi", ..
                 toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1.lis", ..
                 toolbox_dir+c+"demos"+c+"Help"+c+"demo1"+c+"demo1.opt"];
 
TypeNomFichier = ["xxcas","geo","loi","loi","listing","res"]; // "xxcas" is not a valid extension!
impression = 0;
err = importModelMASCARET(id,TabNomFichier,TypeNomFichier,impression);
// assert_checkequal(err,0);
        
// get the error message
if err then 
   [err,message] = getErrorMASCARET(id)
end   
        
// model deletion
err = deleteMASCARET(id);
assert_checkequal(err,0);
